using System;
using System.IO;
using Newtonsoft.Json;

public class KuliahMahasiswa
{
    private string jsonFile;

    public KuliahMahasiswa(string jsonFile)
    {
        this.jsonFile = jsonFile;
    }

    public void ReadJSON()
    {
        string json = File.ReadAllText(jsonFile);
        dynamic data = JsonConvert.DeserializeObject(json);

        Console.WriteLine("Daftar mata kuliah yang diambil:");
        int i = 1;
        foreach (var mk in data.mata_kuliah)
        {
            string kode_mk = mk.kode_mk;
            string nama_mk = mk.nama_mk;
            Console.WriteLine($"MK {i} {kode_mk} - {nama_mk}");
            i++;
        }
    }
}

// Contoh penggunaan
KuliahMahasiswa mahasiswa1 = new KuliahMahasiswa("data.json");
mahasiswa1.ReadJSON();
